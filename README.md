// –§—É–Ω–∫—Ü–∏—è –¥–ª—è —Ä–∞—Å—á–µ—Ç–∞ —Ä–∞—Å—Å—Ç–æ—è–Ω–∏—è –º–µ–∂–¥—É –¥–≤—É–º—è —Ç–æ—á–∫–∞–º–∏ –Ω–∞ –ó–µ–º–ª–µ
float getDistance(float lat1, float lon1, float lat2, float lon2) {
  float lat1Rad = radians(lat1);  // –ü—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏–µ —à–∏—Ä–æ—Ç—ã –≤ —Ä–∞–¥–∏–∞–Ω—ã
  float lon1Rad = radians(lon1);  // –ü—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏–µ –¥–æ–ª–≥–æ—Ç—ã –≤ —Ä–∞–¥–∏–∞–Ω—ã
  float lat2Rad = radians(lat2);
  float lon2Rad = radians(lon2);

  float dlat = lat2Rad - lat1Rad;  // –†–∞–∑–Ω–∏—Ü–∞ –≤ —à–∏—Ä–æ—Ç–µ
  float dlon = lon2Rad - lon1Rad;  // –†–∞–∑–Ω–∏—Ü–∞ –≤ –¥–æ–ª–≥–æ—Ç–µ

  // –§–æ—Ä–º—É–ª–∞ –≥–∞–≤–µ—Ä—Å–∏–Ω—É—Å–æ–≤ –¥–ª—è —Ä–∞—Å—á–µ—Ç–∞ —Ä–∞—Å—Å—Ç–æ—è–Ω–∏—è
  float a = sin(dlat / 2) * sin(dlat / 2) + cos(lat1Rad) * cos(lat2Rad) * sin(dlon / 2) * sin(dlon / 2);
  float c = 2 * atan2(sqrt(a), sqrt(1 - a));
  float distance = 6371 * c;  // –†–∞—Å—Å—Ç–æ—è–Ω–∏–µ –≤ –∫–∏–ª–æ–º–µ—Ç—Ä–∞—Ö

  return distance;
}

// –§—É–Ω–∫—Ü–∏—è –¥–ª—è —Ä–∞—Å—á–µ—Ç–∞ –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–∞–∑–∏–º—É—Ç–∞) –º–µ–∂–¥—É –¥–≤—É–º—è —Ç–æ—á–∫–∞–º–∏
float getBearing(float lat1, float lon1, float lat2, float lon2) {
  float lat1Rad = radians(lat1);  // –ü—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏–µ —à–∏—Ä–æ—Ç—ã –≤ —Ä–∞–¥–∏–∞–Ω—ã
  float lon1Rad = radians(lon1);  // –ü—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏–µ –¥–æ–ª–≥–æ—Ç—ã –≤ —Ä–∞–¥–∏–∞–Ω—ã
  float lat2Rad = radians(lat2);
  float lon2Rad = radians(lon2);

  float dLon = lon2Rad - lon1Rad;  // –†–∞–∑–Ω–∏—Ü–∞ –≤ –¥–æ–ª–≥–æ—Ç–µ

  // –§–æ—Ä–º—É–ª–∞ –¥–ª—è —Ä–∞—Å—á–µ—Ç–∞ –∞–∑–∏–º—É—Ç–∞
  float y = sin(dLon) * cos(lat2Rad);
  float x = cos(lat1Rad) * sin(lat2Rad) - sin(lat1Rad) * cos(lat2Rad) * cos(dLon);

  float bearing = atan2(y, x);  // –ê–∑–∏–º—É—Ç –≤ —Ä–∞–¥–∏–∞–Ω–∞—Ö
  bearing = degrees(bearing);  // –ü—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏–µ –≤ –≥—Ä–∞–¥—É—Å—ã

  if (bearing < 0) bearing += 360;  // –ü—Ä–∏–≤–µ–¥–µ–Ω–∏–µ –∫ –¥–∏–∞–ø–∞–∑–æ–Ω—É 0-360 –≥—Ä–∞–¥—É—Å–æ–≤

  return bearing;
}

void controlMotors(float bearing) {

  // –£–ø—Ä–∞–≤–ª–µ–Ω–∏–µ –º–æ—Ç–æ—Ä–∞–º–∏ –≤ –∑–∞–≤–∏—Å–∏–º–æ—Å—Ç–∏ –æ—Ç –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (bearing)

    motor2.write(baseSpeed);

    motor3.write(baseSpeed + 20);
